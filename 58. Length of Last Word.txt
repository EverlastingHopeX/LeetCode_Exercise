class Solution {
    public int lengthOfLastWord(String s) {
        int i=s.length()-1;
        while(i>=0 && s.charAt(i)==' '){
            i--;
        }
        int end=i;
        for(;i>=0;i--){
            if(s.charAt(i)==' '){
                break;
            }
        }
        return end-i;
    }
}
