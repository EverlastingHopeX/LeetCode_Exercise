class Solution {
    public int search(ArrayReader reader, int target) {
        int low=0;
        int high=20000;
        while(low<=high){
            int mid=low+(high-low)/2;
            int cur=reader.get(mid);
            // System.out.println(low+","+mid+","+high);
            if(cur<target){
                low=mid+1;
            }
            else if(cur>target){
                high=mid-1;
            }
            else{
                return mid;
            }
        }
        return -1;
    }
}
