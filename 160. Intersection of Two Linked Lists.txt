public class Solution {
    public ListNode getIntersectionNode(ListNode headA, ListNode headB) {
        ListNode cur=headA;
        int m=0;
        while(cur!=null){
            m++;
            cur=cur.next;
        }
        cur=headB;
        int n=0;
        while(cur!=null){
            n++;
            cur=cur.next;
        }
        ListNode a=headA;
        ListNode b=headB;
        while(m>n){
            m--;
            a=a.next;
        }
        while(n>m){
            n--;
            b=b.next;
        }
        while(a!=b){
            a=a.next;
            b=b.next;
        }
        return a;
    }
}
