class Solution {
    public int twoSumLessThanK(int[] A, int K) {
        Arrays.sort(A);
        // for(int a:A){
        //     System.out.println(a);
        // }
        int max=-1;
        int left=0;
        int right=A.length-1;
        while(left<right){
            int sum=A[left]+A[right];
            if(sum>=K){
                right--;
            }
            else{
                max=Math.max(sum,max);
                left++;
            }
        }
        return max;
    }
}
